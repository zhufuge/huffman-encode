* Installation

#+BEGIN_SRC sh
$ npm install huffman-encode
#+END_SRC

* Usage

#+BEGIN_SRC js
  const huffmanEncode = require('huffman-encode')

  const probabilities = {
    a: 1,
    b: 2,
    c: 3,
    d: 4,
    e: 5,
    f: 6
  }

  huffmanEncode(probabilities)
  // => {
  //   "a": "1000",
  //   "b": "1001",
  //   "c": "101",
  //   "d": "00",
  //   "e": "01",
  //   "f": "11"
  // }
#+END_SRC
